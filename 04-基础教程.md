### 4.1.2 创建入口文件

创建 `extensions/weather-query/index.ts`：

```typescript
// ========================================
// 导入依赖
// ========================================

// 导入插件 API 类型
import type { OpenClawPluginApi } from "openclaw/plugin-sdk/core";

// 导入 TypeBox 用于定义类型安全的 Schema
import { Type } from "@sinclair/typebox";

// ========================================
// 类型定义
// ========================================

// 插件配置接口
interface WeatherConfig {
  apiKey?: string;       // API 密钥
  defaultCity?: string;  // 默认城市
  units?: "metric" | "imperial";  // 温度单位
}

// 天气响应接口
interface WeatherResponse {
  city: string;          // 城市名
  temperature: number;   // 温度
  condition: string;     // 天气状况
  humidity: number;      // 湿度
  windSpeed: number;     // 风速
}

// ========================================
// 配置 Schema 定义
// ========================================

const configSchema = {
  // 解析配置函数
  parse(value: unknown): WeatherConfig {
    // 检查值是否为有效对象
    const raw =
      value && typeof value === "object" && !Array.isArray(value)
        ? (value as Record<string, unknown>)
        : {};

    // 返回解析后的配置
    return {
      // 获取 API 密钥，支持从环境变量读取
      apiKey: raw.apiKey as string | undefined,
      // 获取默认城市，默认为"北京"
      defaultCity: (raw.defaultCity as string) || "北京",
      // 获取温度单位，默认为公制
      units: (raw.units as "metric" | "imperial") || "metric",
    };
  },

  // JSON Schema 定义
  jsonSchema: {
    type: "object",
    additionalProperties: false,
    properties: {
      apiKey: { type: "string" },
      defaultCity: { type: "string", default: "北京" },
      units: {
        type: "string",
        enum: ["metric", "imperial"],
        default: "metric",
      },
    },
  },

  // UI 提示信息
  uiHints: {
    apiKey: {
      label: "API 密钥",
      sensitive: true,
      placeholder: "输入您的天气 API 密钥",
    },
    defaultCity: {
      label: "默认城市",
      placeholder: "例如：北京",
    },
    units: {
      label: "温度单位",
      help: "metric=摄氏度，imperial=华氏度",
    },
  },
};

// ========================================
// 天气查询函数
// ========================================

/**
 * 查询指定城市的天气
 * @param city 城市名称
 * @param apiKey API 密钥
 * @param units 温度单位
 * @returns 天气信息
 */
async function queryWeather(
  city: string,
  apiKey: string | undefined,
  units: "metric" | "imperial",
): Promise<WeatherResponse> {
  // 如果没有 API 密钥，返回模拟数据
  if (!apiKey) {
    // 返回模拟数据用于演示
    return {
      city,
      temperature: 22,
      condition: "晴朗",
      humidity: 45,
      windSpeed: 3.5,
    };
  }

  // 实际 API 调用（这里使用模拟实现）
  // 在实际项目中，您应该调用真实的天气 API
  // 例如：OpenWeatherMap、和风天气等
  const response = await fetch(
    `https://api.example.com/weather?city=${encodeURIComponent(city)}&appid=${apiKey}&units=${units}`,
  );

  if (!response.ok) {
    throw new Error(`天气查询失败: ${response.statusText}`);
  }

  const data = await response.json();

  return {
    city: data.name,
    temperature: data.main.temp,
    condition: data.weather[0].description,
    humidity: data.main.humidity,
    windSpeed: data.wind.speed,
  };
}

// ========================================
// 插件定义
// ========================================

const weatherPlugin = {
  // 插件 ID
  id: "weather-query",

  // 插件名称
  name: "天气查询",

  // 插件描述
  description: "为 Agent 提供天气查询能力",

  // 配置 Schema
  configSchema,

  // 注册函数
  register(api: OpenClawPluginApi) {
    // 解析配置
    const config = configSchema.parse(api.pluginConfig);

    // 记录日志
    api.logger.info("[weather-query] 插件加载中...");

    // 检查 API 密钥
    if (!config.apiKey) {
      api.logger.warn("[weather-query] 未配置 API 密钥，将使用模拟数据");
    }

    // ========================================
    // 注册天气查询工具
    // ========================================
    api.registerTool({
      // 工具名称（使用 snake_case）
      name: "query_weather",

      // 工具标签
      label: "查询天气",

      // 工具描述
      // Agent 会根据这个描述决定何时调用此工具
      description:
        "查询指定城市的天气信息。当用户询问天气、温度、是否需要带伞等问题时使用此工具。",

      // 参数 Schema
      parameters: Type.Object({
        // 城市参数（必需）
        city: Type.String({
          description: "要查询天气的城市名称，例如：北京、上海、纽约",
        }),
        // 单位参数（可选）
        units: Type.Optional(
          Type.String({
            description: "温度单位：metric（摄氏度）或 imperial（华氏度）",
            enum: ["metric", "imperial"],
          }),
        ),
      }),

      // 执行函数
      async execute(_toolCallId, params) {
        try {
          // 获取参数值，使用默认城市作为后备
          const city = params?.city || config.defaultCity || "北京";
          const units = params?.units || config.units || "metric";

          // 记录查询日志
          api.logger.debug?.(`[weather-query] 查询城市: ${city}`);

          // 执行天气查询
          const weather = await queryWeather(city, config.apiKey, units);

          // 格式化温度显示
          const tempUnit = units === "metric" ? "°C" : "°F";
          const windUnit = units === "metric" ? "米/秒" : "英里/小时";

          // 构建结果文本
          const resultText = [
            `${weather.city} 的天气：`,
            `- 温度：${weather.temperature}${tempUnit}`,
            `- 天气：${weather.condition}`,
            `- 湿度：${weather.humidity}%`,
            `- 风速：${weather.windSpeed} ${windUnit}`,
          ].join("\n");

          // 返回工具结果
          return {
            content: [
              {
                type: "text" as const,
                text: resultText,
              },
            ],
            // 可选：附加详细信息
            details: weather,
          };
        } catch (error) {
          // 错误处理
          const errorMessage =
            error instanceof Error ? error.message : String(error);
          api.logger.error(`[weather-query] 查询失败: ${errorMessage}`);

          // 返回错误信息
          return {
            content: [
              {
                type: "text" as const,
                text: `天气查询失败: ${errorMessage}`,
              },
            ],
          };
        }
      },
    });

    // 记录完成日志
    api.logger.info("[weather-query] 插件已加载");
  },
};

// 导出插件（必须使用 default 导出）
export default weatherPlugin;
```

创建 `extensions/message-logger/index.ts`：

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi } from "openclaw/plugin-sdk/core";

// Node.js 内置模块
import fs from "node:fs";
import path from "node:path";
import os from "node:os";

// ========================================
// 类型定义
// ========================================

interface LoggerConfig {
  logDir: string;
  logFormat: "jsonl" | "text";
  includeMetadata: boolean;
}

interface MessageLogEntry {
  timestamp: string;
  direction: "inbound" | "outbound";
  from?: string;
  to?: string;
  content: string;
  channelId: string;
  success?: boolean;
  metadata?: Record<string, unknown>;
}

// ========================================
// 配置 Schema
// ========================================

const configSchema = {
  parse(value: unknown): LoggerConfig {
    const raw =
      value && typeof value === "object" && !Array.isArray(value)
        ? (value as Record<string, unknown>)
        : {};

    return {
      logDir: (raw.logDir as string) || "~/.openclaw/logs/messages",
      logFormat: (raw.logFormat as "jsonl" | "text") || "jsonl",
      includeMetadata: raw.includeMetadata !== false,
    };
  },
};

// ========================================
// 日志写入器类
// ========================================

class MessageLogWriter {
  private logDir: string;
  private format: "jsonl" | "text";
  private includeMetadata: boolean;
  private currentStream: fs.WriteStream | null = null;
  private currentDate: string = "";

  constructor(config: LoggerConfig) {
    // 展开路径中的 ~ 符号
    this.logDir = config.logDir.replace(/^~/, os.homedir());
    this.format = config.logFormat;
    this.includeMetadata = config.includeMetadata;
  }

  /**
   * 确保日志目录存在
   */
  private ensureLogDir(): void {
    if (!fs.existsSync(this.logDir)) {
      // 递归创建目录
      fs.mkdirSync(this.logDir, { recursive: true });
    }
  }

  /**
   * 获取当前日期字符串（用于日志文件名）
   */
  private getCurrentDate(): string {
    const now = new Date();
    return now.toISOString().split("T")[0]; // YYYY-MM-DD
  }

  /**
   * 获取日志文件路径
   */
  private getLogFilePath(): string {
    const date = this.getCurrentDate();
    const ext = this.format === "jsonl" ? ".jsonl" : ".log";
    return path.join(this.logDir, `messages-${date}${ext}`);
  }

  /**
   * 获取写入流
   */
  private getStream(): fs.WriteStream {
    const date = this.getCurrentDate();

    // 如果日期变了，关闭旧流，创建新流
    if (this.currentDate !== date) {
      if (this.currentStream) {
        this.currentStream.end();
      }
      this.ensureLogDir();
      this.currentStream = fs.createWriteStream(this.getLogFilePath(), {
        flags: "a", // 追加模式
      });
      this.currentDate = date;
    }

    return this.currentStream!;
  }

  /**
   * 格式化日志条目
   */
  private formatEntry(entry: MessageLogEntry): string {
    if (this.format === "jsonl") {
      // JSON 行格式
      return JSON.stringify(entry);
    } else {
      // 纯文本格式
      const parts = [
        `[${entry.timestamp}]`,
        entry.direction === "inbound" ? "←" : "→",
        entry.direction === "inbound"
          ? `从 ${entry.from}`
          : `到 ${entry.to}`,
        `(${entry.channelId})`,
        entry.content.substring(0, 100), // 截断长消息
      ];

      if (entry.success === false) {
        parts.push("[失败]");
      }

      return parts.join(" ");
    }
  }

  /**
   * 写入日志条目
   */
  write(entry: MessageLogEntry): void {
    try {
      const stream = this.getStream();
      const line = this.formatEntry(entry);
      stream.write(line + "\n");
    } catch (error) {
      console.error("[message-logger] 写入日志失败:", error);
    }
  }

  /**
   * 关闭写入流
   */
  close(): void {
    if (this.currentStream) {
      this.currentStream.end();
      this.currentStream = null;
    }
  }
}

// ========================================
// 插件定义
// ========================================

const messageLoggerPlugin = {
  id: "message-logger",
  name: "消息记录器",
  description: "记录所有收发的消息到文件",

  configSchema,

  register(api: OpenClawPluginApi) {
    // 解析配置
    const config = configSchema.parse(api.pluginConfig);

    // 创建日志写入器
    const writer = new MessageLogWriter(config);

    api.logger.info("[message-logger] 插件加载中...");
    api.logger.info(`[message-logger] 日志目录: ${config.logDir}`);
    api.logger.info(`[message-logger] 日志格式: ${config.logFormat}`);

    // ========================================
    // 注册消息接收钩子
    // ========================================
    api.on("message_received", async (event, ctx) => {
      // 构建日志条目
      const entry: MessageLogEntry = {
        timestamp: new Date().toISOString(),
        direction: "inbound",
        from: event.from,
        content: event.content,
        channelId: ctx.channelId,
      };

      // 如果配置了包含元数据，添加额外信息
      if (config.includeMetadata && event.metadata) {
        entry.metadata = event.metadata;
      }

      // 写入日志
      writer.write(entry);
    });

    // ========================================
    // 注册消息发送钩子
    // ========================================
    api.on("message_sent", async (event, ctx) => {
      // 构建日志条目
      const entry: MessageLogEntry = {
        timestamp: new Date().toISOString(),
        direction: "outbound",
        to: event.to,
        content: event.content,
        channelId: ctx.channelId,
        success: event.success,
      };

      // 如果发送失败，记录错误
      if (!event.success && event.error) {
        entry.metadata = { error: event.error };
      }

      // 写入日志
      writer.write(entry);
    });

    // ========================================
    // 注册后台服务（用于清理资源）
    // ========================================
    api.registerService({
      id: "message-logger",

      start() {
        api.logger.info("[message-logger] 服务已启动");
      },

      stop() {
        // 关闭日志写入流
        writer.close();
        api.logger.info("[message-logger] 服务已停止");
      },
    });

    api.logger.info("[message-logger] 插件已加载");
  },
};

export default messageLoggerPlugin;
```

### 4.3.2 创建入口文件

创建 `extensions/status-api/index.ts`：

```typescript
// ========================================
// 导入依赖
// ========================================

import type { IncomingMessage, ServerResponse } from "node:http";
import type { OpenClawPluginApi } from "openclaw/plugin-sdk/core";

// ========================================
// 类型定义
// ========================================

interface StatusApiConfig {
  path: string;
  requireAuth: boolean;
}

// 状态响应接口
interface StatusResponse {
  status: "ok" | "error";
  timestamp: string;
  uptime: number;
  version: string;
  plugins: string[];
}

// ========================================
// 配置 Schema
// ========================================

const configSchema = {
  parse(value: unknown): StatusApiConfig {
    const raw =
      value && typeof value === "object" && !Array.isArray(value)
        ? (value as Record<string, unknown>)
        : {};

    return {
      path: (raw.path as string) || "/plugins/status",
      requireAuth: raw.requireAuth !== false,
    };
  },
};

// ========================================
// 辅助函数
// ========================================

/**
 * 发送 JSON 响应
 */
function sendJson(
  res: ServerResponse,
  statusCode: number,
  data: unknown,
): void {
  // 设置响应头
  res.writeHead(statusCode, {
    "Content-Type": "application/json",
    // 允许跨域请求
    "Access-Control-Allow-Origin": "*",
  });

  // 发送 JSON 响应
  res.end(JSON.stringify(data, null, 2));
}

/**
 * 解析 URL 路径
 */
function parseUrl(req: IncomingMessage): {
  pathname: string;
  query: URLSearchParams;
} {
  const url = new URL(
    req.url || "/",
    `http://${req.headers.host || "localhost"}`,
  );
  return {
    pathname: url.pathname,
    query: url.searchParams,
  };
}

// ========================================
// 插件定义
// ========================================

const statusApiPlugin = {
  id: "status-api",
  name: "状态 API",
  description: "提供系统状态查询的 HTTP API",

  configSchema,

  register(api: OpenClawPluginApi) {
    // 解析配置
    const config = configSchema.parse(api.pluginConfig);

    // 记录启动时间
    const startTime = Date.now();

    api.logger.info("[status-api] 插件加载中...");
    api.logger.info(`[status-api] API 路径: ${config.path}`);

    // ========================================
    // 注册 HTTP 路由
    // ========================================
    api.registerHttpRoute({
      // 路由路径
      path: config.path,

      // 认证模式
      // "gateway" = 需要 Gateway Token
      // "plugin" = 使用插件自身的认证
      auth: config.requireAuth ? "gateway" : "plugin",

      // 匹配模式
      // "exact" = 精确匹配
      // "prefix" = 前缀匹配
      match: "prefix",

      // 路由处理器
      async handler(
        req: IncomingMessage,
        res: ServerResponse,
      ): Promise<boolean> {
        const { pathname, query } = parseUrl(req);

        // 路由到具体的处理函数
        const basePath = config.path;

        // GET /status - 获取系统状态
        if (pathname === basePath && req.method === "GET") {
          return handleGetStatus(req, res, api, startTime);
        }

        // GET /status/health - 健康检查
        if (pathname === `${basePath}/health` && req.method === "GET") {
          return handleHealthCheck(req, res);
        }

        // GET /status/plugins - 获取插件列表
        if (pathname === `${basePath}/plugins` && req.method === "GET") {
          return handleGetPlugins(req, res, api);
        }

        // 未知路由，返回 404
        sendJson(res, 404, {
          status: "error",
          message: "Not Found",
        });
        return true;
      },
    });

    api.logger.info("[status-api] 插件已加载");
  },
};

// ========================================
// 路由处理函数
// ========================================

/**
 * 处理 GET /status 请求
 */
function handleGetStatus(
  _req: IncomingMessage,
  res: ServerResponse,
  api: OpenClawPluginApi,
  startTime: number,
): boolean {
  // 计算运行时间（秒）
  const uptime = Math.floor((Date.now() - startTime) / 1000);

  // 构建状态响应
  const response: StatusResponse = {
    status: "ok",
    timestamp: new Date().toISOString(),
    uptime,
    version: api.version || "unknown",
    plugins: [], // 这里可以添加已加载的插件列表
  };

  // 发送响应
  sendJson(res, 200, response);
  return true;
}

/**
 * 处理 GET /status/health 请求
 */
function handleHealthCheck(
  _req: IncomingMessage,
  res: ServerResponse,
): boolean {
  // 简单的健康检查响应
  sendJson(res, 200, {
    status: "healthy",
    timestamp: new Date().toISOString(),
  });
  return true;
}

/**
 * 处理 GET /status/plugins 请求
 */
function handleGetPlugins(
  _req: IncomingMessage,
  res: ServerResponse,
  api: OpenClawPluginApi,
): boolean {
  // 返回插件信息
  sendJson(res, 200, {
    currentPlugin: {
      id: api.id,
      name: api.name,
      version: api.version,
    },
    // 可以扩展返回更多插件信息
  });
  return true;
}

export default statusApiPlugin;
```

### 4.4.2 创建入口文件

创建 `extensions/notes-cli/index.ts`：

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi } from "openclaw/plugin-sdk/core";
import type { Command } from "commander";

// Node.js 内置模块
import fs from "node:fs";
import path from "node:path";
import os from "node:os";

// ========================================
// 类型定义
// ========================================

interface NotesConfig {
  notesDir: string;
}

interface Note {
  id: string;
  title: string;
  content: string;
  createdAt: string;
  updatedAt: string;
}

// ========================================
// 配置 Schema
// ========================================

const configSchema = {
  parse(value: unknown): NotesConfig {
    const raw =
      value && typeof value === "object" && !Array.isArray(value)
        ? (value as Record<string, unknown>)
        : {};

    return {
      notesDir: (raw.notesDir as string) || "~/.openclaw/notes",
    };
  },
};

// ========================================
// 笔记管理类
// ========================================

class NotesManager {
  private notesDir: string;

  constructor(notesDir: string) {
    // 展开路径中的 ~ 符号
    this.notesDir = notesDir.replace(/^~/, os.homedir());
  }

  /**
   * 确保笔记目录存在
   */
  private ensureDir(): void {
    if (!fs.existsSync(this.notesDir)) {
      fs.mkdirSync(this.notesDir, { recursive: true });
    }
  }

  /**
   * 获取笔记文件路径
   */
  private getNotePath(id: string): string {
    return path.join(this.notesDir, `${id}.json`);
  }

  /**
   * 生成唯一 ID
   */
  private generateId(): string {
    return Date.now().toString(36) + Math.random().toString(36).substr(2);
  }

  /**
   * 列出所有笔记
   */
  listNotes(): Note[] {
    this.ensureDir();

    // 读取目录中的所有 JSON 文件
    const files = fs
      .readdirSync(this.notesDir)
      .filter((f) => f.endsWith(".json"));

    // 读取并解析每个笔记
    return files.map((file) => {
      const content = fs.readFileSync(
        path.join(this.notesDir, file),
        "utf-8",
      );
      return JSON.parse(content) as Note;
    });
  }

  /**
   * 创建新笔记
   */
  createNote(title: string, content: string): Note {
    this.ensureDir();

    const now = new Date().toISOString();
    const note: Note = {
      id: this.generateId(),
      title,
      content,
      createdAt: now,
      updatedAt: now,
    };

    // 写入文件
    fs.writeFileSync(
      this.getNotePath(note.id),
      JSON.stringify(note, null, 2),
    );

    return note;
  }

  /**
   * 读取笔记
   */
  getNote(id: string): Note | null {
    const filePath = this.getNotePath(id);

    if (!fs.existsSync(filePath)) {
      return null;
    }

    const content = fs.readFileSync(filePath, "utf-8");
    return JSON.parse(content) as Note;
  }

  /**
   * 更新笔记
   */
  updateNote(id: string, updates: Partial<Note>): Note | null {
    const note = this.getNote(id);

    if (!note) {
      return null;
    }

    // 应用更新
    const updatedNote: Note = {
      ...note,
      ...updates,
      id: note.id, // ID 不可更改
      createdAt: note.createdAt, // 创建时间不可更改
      updatedAt: new Date().toISOString(),
    };

    // 写入文件
    fs.writeFileSync(
      this.getNotePath(id),
      JSON.stringify(updatedNote, null, 2),
    );

    return updatedNote;
  }

  /**
   * 删除笔记
   */
  deleteNote(id: string): boolean {
    const filePath = this.getNotePath(id);

    if (!fs.existsSync(filePath)) {
      return false;
    }

    fs.unlinkSync(filePath);
    return true;
  }
}

// ========================================
// 插件定义
// ========================================

const notesCliPlugin = {
  id: "notes-cli",
  name: "笔记 CLI",
  description: "提供笔记管理的 CLI 命令",

  configSchema,

  register(api: OpenClawPluginApi) {
    // 解析配置
    const config = configSchema.parse(api.pluginConfig);

    // 创建笔记管理器
    const notesManager = new NotesManager(config.notesDir);

    api.logger.info("[notes-cli] 插件加载中...");

    // ========================================
    // 注册 CLI 命令
    // ========================================
    api.registerCli(
      ({ program }: { program: Command }) => {
        // 创建 notes 命令组
        const notes = program.command("notes").description("笔记管理");

        // notes list - 列出所有笔记
        notes
          .command("list")
          .description("列出所有笔记")
          .option("-j, --json", "以 JSON 格式输出")
          .action((options) => {
            // 获取所有笔记
            const notesList = notesManager.listNotes();

            if (notesList.length === 0) {
              console.log("暂无笔记");
              return;
            }

            // 根据选项输出不同格式
            if (options.json) {
              console.log(JSON.stringify(notesList, null, 2));
            } else {
              console.log("\n笔记列表：\n");
              notesList.forEach((note) => {
                console.log(`  [${note.id}] ${note.title}`);
                console.log(`      创建于: ${note.createdAt}`);
                console.log("");
              });
            }
          });

        // notes create - 创建笔记
        notes
          .command("create")
          .description("创建新笔记")
          .argument("<title>", "笔记标题")
          .argument("[content]", "笔记内容")
          .action((title, content) => {
            // 创建笔记
            const note = notesManager.createNote(title, content || "");

            console.log(`\n笔记已创建：`);
            console.log(`  ID: ${note.id}`);
            console.log(`  标题: ${note.title}`);
          });

        // notes show - 查看笔记
        notes
          .command("show")
          .description("查看笔记详情")
          .argument("<id>", "笔记 ID")
          .action((id) => {
            // 获取笔记
            const note = notesManager.getNote(id);

            if (!note) {
              console.error(`错误：找不到笔记 ${id}`);
              process.exit(1);
            }

            // 显示笔记内容
            console.log(`\n=== ${note.title} ===\n`);
            console.log(note.content || "(无内容)");
            console.log(`\n创建于: ${note.createdAt}`);
            console.log(`更新于: ${note.updatedAt}`);
          });

        // notes edit - 编辑笔记
        notes
          .command("edit")
          .description("编辑笔记")
          .argument("<id>", "笔记 ID")
          .option("-t, --title <title>", "新标题")
          .option("-c, --content <content>", "新内容")
          .action((id, options) => {
            // 检查是否有更新
            if (!options.title && !options.content) {
              console.error("错误：请指定要更新的内容 (--title 或 --content)");
              process.exit(1);
            }

            // 构建更新对象
            const updates: Partial<Note> = {};
            if (options.title) updates.title = options.title;
            if (options.content) updates.content = options.content;

            // 执行更新
            const note = notesManager.updateNote(id, updates);

            if (!note) {
              console.error(`错误：找不到笔记 ${id}`);
              process.exit(1);
            }

            console.log(`笔记 ${id} 已更新`);
          });

        // notes delete - 删除笔记
        notes
          .command("delete")
          .description("删除笔记")
          .argument("<id>", "笔记 ID")
          .option("-f, --force", "强制删除，不确认")
          .action((id, options) => {
            // 检查笔记是否存在
            const note = notesManager.getNote(id);

            if (!note) {
              console.error(`错误：找不到笔记 ${id}`);
              process.exit(1);
            }

            // 如果不是强制删除，需要确认
            if (!options.force) {
              console.log(`即将删除笔记：${note.title}`);
              console.log("使用 --force 选项跳过确认");
              return;
            }

            // 执行删除
            notesManager.deleteNote(id);
            console.log(`笔记 ${id} 已删除`);
          });
      },
      // 指定注册的命令
      { commands: ["notes"] },
    );

    api.logger.info("[notes-cli] 插件已加载");
    api.logger.info(`[notes-cli] 笔记目录: ${config.notesDir}`);
  },
};

export default notesCliPlugin;
```
