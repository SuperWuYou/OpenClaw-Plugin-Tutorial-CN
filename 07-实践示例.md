## 7.1 示例一：翻译工具插件

一个简单的翻译工具插件，演示工具注册和 API 调用。

### 7.1.1 清单文件

```json
  "configSchema": {
    "type": "object",
    "additionalProperties": false,
    "properties": {
      "defaultTargetLang": {
        "type": "string",
        "default": "zh",
        "description": "默认目标语言"

    "defaultTargetLang": {
      "label": "默认目标语言",
      "placeholder": "zh, en, ja...",
      "help": "ISO 639-1 语言代码"
```

### 7.1.2 入口文件

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi } from "openclaw/plugin-sdk/core";
import { Type } from "@sinclair/typebox";

// ========================================
// 类型定义
// ========================================

interface TranslatorConfig {
  defaultTargetLang?: string;
  apiEndpoint?: string;
}

// 语言代码到名称的映射
const LANGUAGE_NAMES: Record<string, string> = {
  zh: "中文",
  en: "英语",
  ja: "日语",
  ko: "韩语",
  fr: "法语",
  de: "德语",
  es: "西班牙语",
  ru: "俄语",
  pt: "葡萄牙语",
  it: "意大利语",
};

// ========================================
// 翻译函数
// ========================================

/**
 * 执行翻译
 * @param text 要翻译的文本
 * @param targetLang 目标语言
 * @param apiEndpoint API 端点（可选）
 * @returns 翻译结果
 */
async function translateText(
  text: string,
  targetLang: string,
  apiEndpoint?: string,
): Promise<string> {
  // 如果没有配置 API 端点，使用模拟翻译
  if (!apiEndpoint) {
    // 模拟翻译延迟
    await new Promise((resolve) => setTimeout(resolve, 500));

    // 返回模拟结果（实际项目中应调用真实 API）
    return `[${LANGUAGE_NAMES[targetLang] || targetLang} 翻译] ${text}`;
  }

  // 调用实际 API
  const response = await fetch(`${apiEndpoint}/translate`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      text,
      target_lang: targetLang,
    }),
  });

  if (!response.ok) {
    throw new Error(`翻译失败: ${response.statusText}`);
  }

  const data = await response.json();
  return data.translated_text;
}

// ========================================
// 插件定义
// ========================================

const translatorPlugin = {
  id: "translator",
  name: "翻译工具",
  description: "为 Agent 提供多语言翻译能力",

  configSchema: {
    parse(value: unknown): TranslatorConfig {
      const raw =
        value && typeof value === "object" && !Array.isArray(value)
          ? (value as Record<string, unknown>)
          : {};

      return {
        defaultTargetLang: (raw.defaultTargetLang as string) || "zh",
        apiEndpoint: raw.apiEndpoint as string | undefined,
      };
    },

    jsonSchema: {
      type: "object",
      additionalProperties: false,
      properties: {
        defaultTargetLang: { type: "string", default: "zh" },
        apiEndpoint: { type: "string" },
      },
    },

    uiHints: {
      defaultTargetLang: {
        label: "默认目标语言",
        placeholder: "zh, en, ja...",
      },
      apiEndpoint: {
        label: "API 端点",
        advanced: true,
      },
    },
  },

  register(api: OpenClawPluginApi) {
    // 解析配置
    const config = this.configSchema.parse(api.pluginConfig);

    api.logger.info("[translator] 插件加载中...");

    // ========================================
    // 注册翻译工具
    // ========================================
    api.registerTool({
      name: "translate",
      label: "翻译",
      description:
        "将文本翻译成指定语言。当用户需要翻译内容、转换语言时使用此工具。",

      parameters: Type.Object({
        // 要翻译的文本
        text: Type.String({
          description: "要翻译的文本内容",
        }),
        // 目标语言
        target_lang: Type.String({
          description: "目标语言代码，如 zh(中文)、en(英语)、ja(日语)",
          default: config.defaultTargetLang || "zh",
        }),
        // 源语言（可选）
        source_lang: Type.Optional(
          Type.String({
            description: "源语言代码（可选，自动检测）",
          }),
        ),
      }),

      async execute(_toolCallId, params) {
        try {
          // 获取参数
          const text = params?.text;
          const targetLang = params?.target_lang || config.defaultTargetLang || "zh";
          const sourceLang = params?.source_lang;

          // 验证参数
          if (!text || typeof text !== "string") {
            return {
              content: [
                {
                  type: "text" as const,
                  text: "错误：请提供要翻译的文本",
                },
              ],
            };
          }

          // 检查文本长度
          if (text.length > 5000) {
            return {
              content: [
                {
                  type: "text" as const,
                  text: "错误：文本过长，最多支持 5000 个字符",
                },
              ],
            };
          }

          // 记录日志
          api.logger.debug?.(
            `[translator] 翻译文本: ${text.slice(0, 50)}... -> ${targetLang}`,
          );

          // 执行翻译
          const translatedText = await translateText(
            text,
            targetLang,
            config.apiEndpoint,
          );

          // 获取语言名称
          const targetLangName = LANGUAGE_NAMES[targetLang] || targetLang;
          const sourceInfo = sourceLang
            ? `从${LANGUAGE_NAMES[sourceLang] || sourceLang}`
            : "自动检测";

          // 返回结果
          return {
            content: [
              {
                type: "text" as const,
                text: `翻译结果（${sourceInfo} → ${targetLangName}）：\n\n${translatedText}`,
              },
            ],
            details: {
              sourceLang,
              targetLang,
              originalText: text,
              translatedText,
            },
          };
        } catch (error) {
          const errorMessage =
            error instanceof Error ? error.message : String(error);

          api.logger.error(`[translator] 翻译失败: ${errorMessage}`);

          return {
            content: [
              {
                type: "text" as const,
                text: `翻译失败: ${errorMessage}`,
              },
            ],
          };
        }
      },
    });

    // ========================================
    // 注册语言检测工具
    // ========================================
    api.registerTool(
      {
        name: "detect_language",
        label: "检测语言",
        description: "检测文本的语言。当需要知道文本是什么语言时使用。",

        parameters: Type.Object({
          text: Type.String({
            description: "要检测语言的文本",
          }),
        }),

        async execute(_toolCallId, params) {
          const text = params?.text;

          if (!text) {
            return {
              content: [
                { type: "text" as const, text: "请提供要检测的文本" },
              ],
            };
          }

          // 简单的语言检测逻辑（实际项目应使用专业库）
          let detectedLang = "unknown";
          let confidence = 0;

          // 检测中文
          if (/[\u4e00-\u9fff]/.test(text)) {
            detectedLang = "zh";
            confidence = 0.9;
          }
          // 检测日文
          else if (/[\u3040-\u309f\u30a0-\u30ff]/.test(text)) {
            detectedLang = "ja";
            confidence = 0.9;
          }
          // 检测韩文
          else if (/[\uac00-\ud7af]/.test(text)) {
            detectedLang = "ko";
            confidence = 0.9;
          }
          // 检测俄文
          else if (/[\u0400-\u04ff]/.test(text)) {
            detectedLang = "ru";
            confidence = 0.9;
          }
          // 默认英文
          else if (/^[a-zA-Z\s.,!?'"()-]+$/.test(text)) {
            detectedLang = "en";
            confidence = 0.7;
          }

          const langName = LANGUAGE_NAMES[detectedLang] || detectedLang;

          return {
            content: [
              {
                type: "text" as const,
                text: `检测到语言: ${langName} (${detectedLang})\n置信度: ${(confidence * 100).toFixed(0)}%`,
              },
            ],
            details: {
              language: detectedLang,
              languageName: langName,
              confidence,
            },
          };
        },
      },
      { optional: true }, // 标记为可选工具
    );

    api.logger.info("[translator] 插件已加载");
  },
};

export default translatorPlugin;
```

### 7.2.2 入口文件

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi } from "openclaw/plugin-sdk/core";
import fs from "node:fs";
import path from "node:path";
import os from "node:os";

// ========================================
// 类型定义
// ========================================

interface StatsConfig {
  statsFile: string;
  trackUsers: boolean;
}

interface MessageStats {
  totalReceived: number;
  totalSent: number;
  byChannel: Record<string, { received: number; sent: number }>;
  byUser: Record<string, { received: number; sent: number }>;
  dailyStats: Record<string, { received: number; sent: number }>;
  lastUpdated: string;
}

// ========================================
// 统计管理类
// ========================================

class StatsManager {
  private statsFile: string;
  private stats: MessageStats;
  private trackUsers: boolean;

  constructor(config: StatsConfig) {
    // 展开路径
    this.statsFile = config.statsFile.replace(/^~/, os.homedir());
    this.trackUsers = config.trackUsers;

    // 初始化统计数据
    this.stats = this.loadStats();
  }

  /**
   * 加载统计数据
   */
  private loadStats(): MessageStats {
    try {
      if (fs.existsSync(this.statsFile)) {
        const content = fs.readFileSync(this.statsFile, "utf-8");
        return JSON.parse(content);
      }
    } catch (error) {
      console.error("[message-stats] 加载统计数据失败:", error);
    }

    // 返回默认统计数据
    return {
      totalReceived: 0,
      totalSent: 0,
      byChannel: {},
      byUser: {},
      dailyStats: {},
      lastUpdated: new Date().toISOString(),
    };
  }

  /**
   * 保存统计数据
   */
  private saveStats(): void {
    try {
      // 确保目录存在
      const dir = path.dirname(this.statsFile);
      if (!fs.existsSync(dir)) {
        fs.mkdirSync(dir, { recursive: true });
      }

      // 更新时间戳
      this.stats.lastUpdated = new Date().toISOString();

      // 写入文件
      fs.writeFileSync(this.statsFile, JSON.stringify(this.stats, null, 2));
    } catch (error) {
      console.error("[message-stats] 保存统计数据失败:", error);
    }
  }

  /**
   * 记录收到的消息
   */
  recordReceived(channelId: string, from?: string): void {
    // 更新总数
    this.stats.totalReceived++;

    // 更新渠道统计
    if (!this.stats.byChannel[channelId]) {
      this.stats.byChannel[channelId] = { received: 0, sent: 0 };
    }
    this.stats.byChannel[channelId].received++;

    // 更新用户统计
    if (this.trackUsers && from) {
      if (!this.stats.byUser[from]) {
        this.stats.byUser[from] = { received: 0, sent: 0 };
      }
      this.stats.byUser[from].received++;
    }

    // 更新每日统计
    const today = new Date().toISOString().split("T")[0];
    if (!this.stats.dailyStats[today]) {
      this.stats.dailyStats[today] = { received: 0, sent: 0 };
    }
    this.stats.dailyStats[today].received++;

    // 保存
    this.saveStats();
  }

  /**
   * 记录发送的消息
   */
  recordSent(channelId: string, to?: string): void {
    // 更新总数
    this.stats.totalSent++;

    // 更新渠道统计
    if (!this.stats.byChannel[channelId]) {
      this.stats.byChannel[channelId] = { received: 0, sent: 0 };
    }
    this.stats.byChannel[channelId].sent++;

    // 更新用户统计
    if (this.trackUsers && to) {
      if (!this.stats.byUser[to]) {
        this.stats.byUser[to] = { received: 0, sent: 0 };
      }
      this.stats.byUser[to].sent++;
    }

    // 更新每日统计
    const today = new Date().toISOString().split("T")[0];
    if (!this.stats.dailyStats[today]) {
      this.stats.dailyStats[today] = { received: 0, sent: 0 };
    }
    this.stats.dailyStats[today].sent++;

    // 保存
    this.saveStats();
  }

  /**
   * 获取统计数据
   */
  getStats(): MessageStats {
    return { ...this.stats };
  }

  /**
   * 重置统计数据
   */
  resetStats(): void {
    this.stats = {
      totalReceived: 0,
      totalSent: 0,
      byChannel: {},
      byUser: {},
      dailyStats: {},
      lastUpdated: new Date().toISOString(),
    };
    this.saveStats();
  }
}

// ========================================
// 插件定义
// ========================================

const messageStatsPlugin = {
  id: "message-stats",
  name: "消息统计",
  description: "统计消息收发情况",

  configSchema: {
    parse(value: unknown): StatsConfig {
      const raw =
        value && typeof value === "object" && !Array.isArray(value)
          ? (value as Record<string, unknown>)
          : {};

      return {
        statsFile: (raw.statsFile as string) || "~/.openclaw/data/message-stats.json",
        trackUsers: raw.trackUsers !== false,
      };
    },

    jsonSchema: {
      type: "object",
      additionalProperties: false,
      properties: {
        statsFile: {
          type: "string",
          default: "~/.openclaw/data/message-stats.json",
        },
        trackUsers: { type: "boolean", default: true },
      },
    },
  },

  register(api: OpenClawPluginApi) {
    // 解析配置
    const config = this.configSchema.parse(api.pluginConfig);

    // 创建统计管理器
    const statsManager = new StatsManager(config);

    api.logger.info("[message-stats] 插件加载中...");

    // ========================================
    // 注册消息钩子
    // ========================================

    // 收到消息钩子
    api.on("message_received", async (event, ctx) => {
      statsManager.recordReceived(ctx.channelId, event.from);
      api.logger.debug?.(
        `[message-stats] 收到消息 - 渠道: ${ctx.channelId}, 发送者: ${event.from}`,
      );
    });

    // 发送消息钩子
    api.on("message_sent", async (event, ctx) => {
      statsManager.recordSent(ctx.channelId, event.to);
      api.logger.debug?.(
        `[message-stats] 消息已发送 - 渠道: ${ctx.channelId}, 接收者: ${event.to}`,
      );
    });

    // ========================================
    // 注册统计命令
    // ========================================
    api.registerCommand({
      name: "stats",
      description: "查看消息统计",

      handler: (ctx) => {
        const stats = statsManager.getStats();

        // 构建统计报告
        const lines = [
          "📊 **消息统计报告**",
          "",
          `总计收到: ${stats.totalReceived} 条`,
          `总计发送: ${stats.totalSent} 条`,
          "",
          "**按渠道统计:**",
        ];

        // 添加渠道统计
        for (const [channel, data] of Object.entries(stats.byChannel)) {
          lines.push(`  ${channel}: 收 ${data.received} / 发 ${data.sent}`);
        }

        // 添加今日统计
        const today = new Date().toISOString().split("T")[0];
        const todayStats = stats.dailyStats[today];
        if (todayStats) {
          lines.push("");
          lines.push(`**今日统计 (${today}):**`);
          lines.push(`  收到: ${todayStats.received} 条`);
          lines.push(`  发送: ${todayStats.sent} 条`);
        }

        return { text: lines.join("\n") };
      },
    });

    // ========================================
    // 注册重置统计命令
    // ========================================
    api.registerCommand({
      name: "stats-reset",
      description: "重置消息统计",
      requireAuth: true, // 需要授权用户

      handler: (ctx) => {
        if (!ctx.isAuthorizedSender) {
          return { text: "错误：需要授权用户才能重置统计" };
        }

        statsManager.resetStats();
        return { text: "✅ 消息统计已重置" };
      },
    });

    // ========================================
    // 注册 Gateway RPC 方法
    // ========================================
    api.registerGatewayMethod(
      "message-stats.get",
      async ({ respond }) => {
        const stats = statsManager.getStats();
        respond(true, stats);
      },
    );

    api.registerGatewayMethod(
      "message-stats.reset",
      async ({ respond, params }) => {
        // 验证确认参数
        if (params?.confirm !== true) {
          respond(false, { error: "需要 confirm: true 参数" });
          return;
        }

        statsManager.resetStats();
        respond(true, { message: "统计已重置" });
      },
    );

    api.logger.info("[message-stats] 插件已加载");
  },
};

export default messageStatsPlugin;
```

### 7.3.2 入口文件

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi, OpenClawPluginServiceContext } from "openclaw/plugin-sdk/core";
import os from "node:os";

// ========================================
// 类型定义
// ========================================

interface HealthCheckConfig {
  intervalMinutes: number;
  checkMemory: boolean;
  memoryThreshold: number;
}

interface HealthStatus {
  timestamp: string;
  status: "healthy" | "warning" | "error";
  memoryUsage?: {
    used: number;
    total: number;
    percentage: number;
  };
  cpuLoad?: number[];
  uptime: number;
  issues: string[];
}

// ========================================
// 健康检查器类
// ========================================

class HealthChecker {
  private config: HealthCheckConfig;
  private logger: PluginLogger;
  private interval: NodeJS.Timeout | null = null;
  private lastStatus: HealthStatus | null = null;

  constructor(config: HealthCheckConfig, logger: PluginLogger) {
    this.config = config;
    this.logger = logger;
  }

  /**
   * 启动定期检查
   */
  start(): void {
    this.logger.info("[health-check] 启动健康检查服务...");

    // 立即执行一次检查
    this.performCheck();

    // 设置定时检查
    this.interval = setInterval(
      () => this.performCheck(),
      this.config.intervalMinutes * 60 * 1000,
    );
  }

  /**
   * 停止检查
   */
  stop(): void {
    this.logger.info("[health-check] 停止健康检查服务...");

    if (this.interval) {
      clearInterval(this.interval);
      this.interval = null;
    }
  }

  /**
   * 执行健康检查
   */
  private performCheck(): void {
    const issues: string[] = [];
    let status: "healthy" | "warning" | "error" = "healthy";

    // 获取内存使用情况
    const totalMemory = os.totalmem();
    const freeMemory = os.freemem();
    const usedMemory = totalMemory - freeMemory;
    const memoryPercentage = (usedMemory / totalMemory) * 100;

    // 检查内存阈值
    if (this.config.checkMemory && memoryPercentage > this.config.memoryThreshold) {
      issues.push(
        `内存使用率过高: ${memoryPercentage.toFixed(1)}% (阈值: ${this.config.memoryThreshold}%)`,
      );
      status = "warning";
    }

    // 如果内存使用超过 95%，标记为错误
    if (memoryPercentage > 95) {
      status = "error";
    }

    // 获取 CPU 负载
    const cpuLoad = os.loadavg();

    // 构建状态对象
    this.lastStatus = {
      timestamp: new Date().toISOString(),
      status,
      memoryUsage: {
        used: usedMemory,
        total: totalMemory,
        percentage: memoryPercentage,
      },
      cpuLoad,
      uptime: os.uptime(),
      issues,
    };

    // 记录日志
    if (status === "healthy") {
      this.logger.debug?.(
        `[health-check] 系统健康 - 内存: ${memoryPercentage.toFixed(1)}%`,
      );
    } else if (status === "warning") {
      this.logger.warn(`[health-check] 系统警告: ${issues.join(", ")}`);
    } else {
      this.logger.error(`[health-check] 系统错误: ${issues.join(", ")}`);
    }
  }

  /**
   * 获取当前状态
   */
  getStatus(): HealthStatus | null {
    return this.lastStatus;
  }
}

// ========================================

const healthCheckPlugin = {
  id: "health-check",
  name: "健康检查服务",
  description: "定期检查系统健康状态",

  configSchema: {
    parse(value: unknown): HealthCheckConfig {
      const raw =
        value && typeof value === "object" && !Array.isArray(value)
          ? (value as Record<string, unknown>)
          : {};

      return {
        intervalMinutes: (raw.intervalMinutes as number) ?? 5,
        checkMemory: raw.checkMemory !== false,
        memoryThreshold: (raw.memoryThreshold as number) ?? 90,
      };
    },

    jsonSchema: {
      type: "object",
      additionalProperties: false,
      properties: {
        intervalMinutes: {
          type: "number",
          default: 5,
          minimum: 1,
          maximum: 60,
        },
        checkMemory: { type: "boolean", default: true },
        memoryThreshold: {
          type: "number",
          default: 90,
          minimum: 50,
          maximum: 100,
        },
      },
    },
  },

  register(api: OpenClawPluginApi) {
    const config = this.configSchema.parse(api.pluginConfig);

    // 创建健康检查器（延迟初始化）
    let healthChecker: HealthChecker | null = null;

    api.logger.info("[health-check] 插件加载中...");

    // ========================================
    // 注册后台服务
    // ========================================
    api.registerService({
      id: "health-check",

      start(ctx: OpenClawPluginServiceContext) {
        healthChecker = new HealthChecker(config, ctx.logger);
        healthChecker.start();
      },

      stop() {
        if (healthChecker) {
          healthChecker.stop();
          healthChecker = null;
        }
      },
    });

    // ========================================
    // 注册健康状态命令
    // ========================================
    api.registerCommand({
      name: "health",
      description: "查看系统健康状态",

      handler: () => {
        if (!healthChecker) {
          return { text: "健康检查服务未启动" };
        }

        const status = healthChecker.getStatus();
        if (!status) {
          return { text: "尚无健康检查数据" };
        }

        // 格式化输出
        const lines = [
          `🏥 **系统健康状态**: ${status.status.toUpperCase()}`,
          "",
          `⏰ 检查时间: ${status.timestamp}`,
          `⌛ 系统运行时间: ${formatUptime(status.uptime)}`,
          "",
        ];

        if (status.memoryUsage) {
          lines.push("💾 **内存使用**:");
          lines.push(
            `  已用: ${formatBytes(status.memoryUsage.used)} / ${formatBytes(status.memoryUsage.total)}`,
          );
          lines.push(`  使用率: ${status.memoryUsage.percentage.toFixed(1)}%`);
          lines.push("");
        }

        if (status.cpuLoad) {
          lines.push("⚙️ **CPU 负载**:");
          lines.push(`  1分钟: ${status.cpuLoad[0].toFixed(2)}`);
          lines.push(`  5分钟: ${status.cpuLoad[1].toFixed(2)}`);
          lines.push(`  15分钟: ${status.cpuLoad[2].toFixed(2)}`);
          lines.push("");
        }

        if (status.issues.length > 0) {
          lines.push("⚠️ **问题**:");
          status.issues.forEach((issue) => lines.push(`  - ${issue}`));
        }

        return { text: lines.join("\n") };
      },
    });

    // ========================================
    // 注册 Gateway RPC 方法
    // ========================================
    api.registerGatewayMethod(
      "health-check.status",
      async ({ respond }) => {
        if (!healthChecker) {
          respond(false, { error: "健康检查服务未启动" });
          return;
        }

        const status = healthChecker.getStatus();
        respond(true, status);
      },
    );

    api.logger.info("[health-check] 插件已加载");
  },
};

// 辅助函数
// ========================================

/**
 * 格式化字节数
 */
function formatBytes(bytes: number): string {
  const units = ["B", "KB", "MB", "GB", "TB"];
  let unitIndex = 0;
  let value = bytes;

  while (value >= 1024 && unitIndex < units.length - 1) {
    value /= 1024;
    unitIndex++;
  }

  return `${value.toFixed(1)} ${units[unitIndex]}`;
}

/**
 * 格式化运行时间
 */
function formatUptime(seconds: number): string {
  const days = Math.floor(seconds / 86400);
  const hours = Math.floor((seconds % 86400) / 3600);
  const minutes = Math.floor((seconds % 3600) / 60);

  const parts: string[] = [];
  if (days > 0) parts.push(`${days}天`);
  if (hours > 0) parts.push(`${hours}小时`);
  parts.push(`${minutes}分钟`);

  return parts.join(" ");
}

export default healthCheckPlugin;
```

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi } from "openclaw/plugin-sdk/core";
import type { IncomingMessage, ServerResponse } from "node:http";
import crypto from "node:crypto";

// ========================================
// 类型定义
// ========================================

interface WebhookConfig {
  secret?: string;
  allowedSources?: string[];
}

interface WebhookPayload {
  source: string;
  event: string;
  data: Record<string, unknown>;
  timestamp: number;
}

// ========================================
// 插件定义
// ========================================

const webhookTriggerPlugin = {
  id: "webhook-trigger",
  name: "Webhook 触发器",
  description: "接收外部 Webhook 并触发 Agent 操作",

  configSchema: {
    parse(value: unknown): WebhookConfig {
      const raw =
        value && typeof value === "object" && !Array.isArray(value)
          ? (value as Record<string, unknown>)
          : {};

      return {
        secret: raw.secret as string | undefined,
        allowedSources: raw.allowedSources as string[] | undefined,
      };
    },

    jsonSchema: {
      type: "object",
      additionalProperties: false,
      properties: {
        secret: { type: "string" },
        allowedSources: {
          type: "array",
          items: { type: "string" },
        },
      },
    },

    uiHints: {
      secret: {
        label: "签名密钥",
        sensitive: true,
      },
      allowedSources: {
        label: "允许的来源",
        help: "IP 地址或域名列表",
      },
    },
  },

  register(api: OpenClawPluginApi) {
    const config = this.configSchema.parse(api.pluginConfig);

    api.logger.info("[webhook-trigger] 插件加载中...");

    // ========================================
    // 注册 HTTP 路由
    // ========================================
    api.registerHttpRoute({
      path: "/webhook/trigger",
      auth: "plugin",
      match: "exact",

      handler: async (req: IncomingMessage, res: ServerResponse) => {
        // 只接受 POST 请求
        if (req.method !== "POST") {
          res.statusCode = 405;
          res.setHeader("Content-Type", "application/json");
          res.end(JSON.stringify({ error: "Method not allowed" }));
          return true;
        }

        try {
          // 读取请求体
          const body = await readRequestBody(req);

          // 验证签名（如果配置了密钥）
          if (config.secret) {
            const signature = req.headers["x-signature"] as string;
            const expectedSignature = crypto
              .createHmac("sha256", config.secret)
              .update(body)
              .digest("hex");

            if (signature !== expectedSignature) {
              res.statusCode = 401;
              res.setHeader("Content-Type", "application/json");
              res.end(JSON.stringify({ error: "Invalid signature" }));
              return true;
            }
          }

          // 解析 payload
          const payload: WebhookPayload = JSON.parse(body);

          // 验证来源（如果配置了允许列表）
          if (config.allowedSources && config.allowedSources.length > 0) {
            const clientIp = req.socket.remoteAddress;
            if (!config.allowedSources.includes(clientIp || "")) {
              res.statusCode = 403;
              res.setHeader("Content-Type", "application/json");
              res.end(JSON.stringify({ error: "Source not allowed" }));
              return true;
            }
          }

          // 记录日志
          api.logger.info(
            `[webhook-trigger] 收到 Webhook: ${payload.source} - ${payload.event}`,
          );

          // 处理 Webhook
          // 这里可以根据 payload 触发不同的操作
          // 例如：发送消息、调用工具等

          // 返回成功响应
          res.statusCode = 200;
          res.setHeader("Content-Type", "application/json");
          res.end(
            JSON.stringify({
              received: true,
              eventId: `evt_${Date.now()}`,
            }),
          );

          return true;
        } catch (error) {
          const errorMessage =
            error instanceof Error ? error.message : String(error);

          api.logger.error(`[webhook-trigger] 处理错误: ${errorMessage}`);

          res.statusCode = 500;
          res.setHeader("Content-Type", "application/json");
          res.end(JSON.stringify({ error: "Internal server error" }));
          return true;
        }
      },
    });

    api.logger.info("[webhook-trigger] 插件已加载");
    api.logger.info("[webhook-trigger] Webhook URL: /plugins/webhook-trigger");
  },
};

// ========================================
// 辅助函数
// ========================================

/**
 * 读取请求体
 */
async function readRequestBody(req: IncomingMessage): Promise<string> {
  return new Promise((resolve, reject) => {
    let body = "";

    req.on("data", (chunk) => {
      body += chunk.toString();
    });

    req.on("end", () => {
      resolve(body);
    });

    req.on("error", reject);
  });
}

export default webhookTriggerPlugin;
```
