## 5.1 开发渠道插件

渠道插件允许 OpenClaw 连接到新的消息平台。这是最复杂的插件类型，需要实现多个接口。

## 5.2 开发提供商插件

提供商插件用于添加新的 AI 模型提供商的认证支持。

### 5.2.1 OAuth 认证示例

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi, ProviderAuthContext } from "openclaw/plugin-sdk/core";
import { buildOauthProviderAuthResult } from "openclaw/plugin-sdk/core";

// ========================================
// 常量定义
// ========================================

const PROVIDER_ID = "acme-ai";
const PROVIDER_LABEL = "Acme AI";
const DEFAULT_MODEL = "acme-ai/advanced-1";

// OAuth 配置
const OAUTH_CONFIG = {
  clientId: process.env.ACME_CLIENT_ID,
  clientSecret: process.env.ACME_CLIENT_SECRET,
  authorizeUrl: "https://auth.acme.ai/authorize",
  tokenUrl: "https://auth.acme.ai/token",
  scopes: ["read", "write", "models"],
};

// ========================================
// OAuth 登录流程
// ========================================

async function loginWithOAuth(ctx: ProviderAuthContext) {
  // 创建进度指示器
  const spin = ctx.prompter.progress("正在启动 OAuth 流程...");

  try {
    // 创建 OAuth 处理器
    const handlers = ctx.oauth.createVpsAwareHandlers({
      clientId: OAUTH_CONFIG.clientId!,
      clientSecret: OAUTH_CONFIG.clientSecret!,
      authorizeUrl: OAUTH_CONFIG.authorizeUrl,
      tokenUrl: OAUTH_CONFIG.tokenUrl,
      scopes: OAUTH_CONFIG.scopes,
    });

    // 生成授权 URL
    const { state, codeVerifier, url } = handlers.createAuthUrl();

    // 在远程环境显示 URL，本地自动打开
    await ctx.openUrl(url);

    // 等待回调
    spin.message("等待授权...");
    const { code } = await handlers.waitForCallback(state);

    // 交换令牌
    spin.message("获取访问令牌...");
    const tokens = await handlers.exchangeCode(code, codeVerifier);

    // 获取用户信息
    spin.message("获取账户信息...");
    const userInfo = await fetchUserInfo(tokens.access_token);

    spin.stop("OAuth 流程完成");

    // 返回认证结果
    return buildOauthProviderAuthResult({
      providerId: PROVIDER_ID,
      defaultModel: DEFAULT_MODEL,
      access: tokens.access_token,
      refresh: tokens.refresh_token,
      expires: tokens.expires_at,
      email: userInfo.email,
      notes: [
        "您的 Acme AI 账户已连接",
        "默认模型设置为 acme-ai/advanced-1",
      ],
    });
  } catch (error) {
    spin.stop("OAuth 流程失败");
    throw error;
  }
}

// ========================================
// API Key 认证示例
// ========================================

async function loginWithApiKey(ctx: ProviderAuthContext) {
  // 询问 API Key
  const apiKey = await ctx.prompter.text({
    message: "请输入您的 Acme AI API 密钥",
    placeholder: "acme-...",
    validate: (value) => {
      if (!value || value.length < 10) {
        return "请输入有效的 API 密钥";
      }
      return true;
    },
  });

  if (!apiKey) {
    throw new Error("需要 API 密钥");
  }

  // 验证 API Key
  const spin = ctx.prompter.progress("验证 API 密钥...");
  try {
    const userInfo = await validateApiKey(apiKey);
    spin.stop("API 密钥验证成功");

    return {
      profiles: [
        {
          profileId: `${PROVIDER_ID}:default`,
          credential: {
            type: "api_key",
            provider: PROVIDER_ID,
            apiKey,
          },
        },
      ],
      defaultModel: DEFAULT_MODEL,
      notes: ["API 密钥已保存"],
    };
  } catch (error) {
    spin.stop("验证失败");
    throw new Error(`API 密钥无效: ${error.message}`);
  }
}

// ========================================
// 插件定义
// ========================================

const acmeProviderPlugin = {
  id: "acme-auth",
  name: "Acme AI Provider Auth",
  description: "Acme AI 模型提供商认证",

  configSchema: {
    parse: (v) => v || {},
    jsonSchema: {
      type: "object",
      additionalProperties: false,
      properties: {},
    },
  },

  register(api: OpenClawPluginApi) {
    // 注册提供商
    api.registerProvider({
      id: PROVIDER_ID,
      label: PROVIDER_LABEL,
      docsPath: "/providers/acme",
      aliases: ["acme"],
      envVars: ["ACME_CLIENT_ID", "ACME_CLIENT_SECRET", "ACME_API_KEY"],

      // 支持的认证方法
      auth: [
        {
          id: "oauth",
          label: "OAuth 登录",
          hint: "推荐：通过浏览器授权",
          kind: "oauth",
          run: loginWithOAuth,
        },
        {
          id: "api-key",
          label: "API 密钥",
          hint: "直接使用 API 密钥",
          kind: "api_key",
          run: loginWithApiKey,
        },
      ],

      // 格式化 API Key 显示
      formatApiKey(cred) {
        if (cred.type === "api_key" && cred.apiKey) {
          // 只显示前8个字符
          return cred.apiKey.slice(0, 8) + "...";
        }
        return undefined;
      },

      // 刷新 OAuth 令牌
      async refreshOAuth(cred) {
        if (!cred.refresh) {
          throw new Error("没有刷新令牌");
        }

        const response = await fetch(OAUTH_CONFIG.tokenUrl, {
          method: "POST",
          headers: {
            "Content-Type": "application/x-www-form-urlencoded",
          },
          body: new URLSearchParams({
            grant_type: "refresh_token",
            refresh_token: cred.refresh,
            client_id: OAUTH_CONFIG.clientId!,
            client_secret: OAUTH_CONFIG.clientSecret!,
          }),
        });

        if (!response.ok) {
          throw new Error("令牌刷新失败");
        }

        const data = await response.json();

        return {
          ...cred,
          access: data.access_token,
          refresh: data.refresh_token ?? cred.refresh,
          expires: Date.now() + data.expires_in * 1000,
        };
      },
    });

    api.logger.info("[acme-auth] 提供商插件已加载");
  },
};

export default acmeProviderPlugin;
```

## 5.3 内存插件开发

内存插件提供 Agent 的长期记忆能力。

### 5.3.1 内存插件结构

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi } from "openclaw/plugin-sdk/core";

// ========================================
// 内存插件定义
// ========================================

const myMemoryPlugin = {
  id: "my-memory",
  name: "My Memory",
  description: "自定义内存存储插件",
  kind: "memory", // 重要：标记为内存类型

  configSchema: {
    parse: (v) => v || {},
    jsonSchema: {
      type: "object",
      additionalProperties: false,
      properties: {
        storagePath: {
          type: "string",
          default: "~/.openclaw/memory",
        },
      },
    },
  },

  register(api: OpenClawPluginApi) {
    // 使用运行时工具工厂创建内存工具
    api.registerTool(
      (ctx) => {
        // 创建搜索工具
        const searchTool = api.runtime.tools.createMemorySearchTool({
          config: ctx.config,
          agentSessionKey: ctx.sessionKey,
        });

        // 创建获取工具
        const getTool = api.runtime.tools.createMemoryGetTool({
          config: ctx.config,
          agentSessionKey: ctx.sessionKey,
        });

        // 如果任一工具创建失败，返回 null
        if (!searchTool || !getTool) {
          return null;
        }

        // 返回工具数组
        return [searchTool, getTool];
      },
      {
        // 指定工具名称
        names: ["memory_search", "memory_get"],
      },
    );

    // 注册 CLI 命令
    api.registerCli(
      ({ program }) => {
        api.runtime.tools.registerMemoryCli(program);
      },
      { commands: ["memory"] },
    );

    api.logger.info("[my-memory] 内存插件已加载");
  },
};

export default myMemoryPlugin;
```

## 5.4 HTTP 服务开发

### 5.4.1 完整的 HTTP 路由示例

```typescript
// ========================================
// 导入依赖
// ========================================

import type { OpenClawPluginApi, OpenClawPluginHttpRouteHandler } from "openclaw/plugin-sdk/core";
import type { IncomingMessage, ServerResponse } from "node:http";

// ========================================
// 类型定义
// ========================================

interface WebhookPayload {
  event: string;
  data: Record<string, unknown>;
  timestamp: number;
}

// ========================================
// HTTP 处理器
// ========================================

/**
 * 创建 Webhook 处理器
 */
function createWebhookHandler(
  logger: PluginLogger,
): OpenClawPluginHttpRouteHandler {
  return async (req: IncomingMessage, res: ServerResponse) => {
    // 只接受 POST 请求
    if (req.method !== "POST") {
      res.statusCode = 405;
      res.end(JSON.stringify({ error: "Method not allowed" }));
      return true;
    }

    try {
      // 解析请求体
      const body = await parseRequestBody(req);

      // 验证签名
      const signature = req.headers["x-signature"] as string;
      if (!verifySignature(body, signature)) {
        res.statusCode = 401;
        res.end(JSON.stringify({ error: "Invalid signature" }));
        return true;
      }

      // 处理事件
      const payload = JSON.parse(body) as WebhookPayload;
      logger.info(`[webhook] 收到事件: ${payload.event}`);

      // 根据事件类型处理
      switch (payload.event) {
        case "message.created":
          await handleMessageCreated(payload.data);
          break;
        case "user.joined":
          await handleUserJoined(payload.data);
          break;
        default:
          logger.warn(`[webhook] 未知事件: ${payload.event}`);
      }

      // 返回成功响应
      res.statusCode = 200;
      res.setHeader("Content-Type", "application/json");
      res.end(JSON.stringify({ received: true }));
      return true;
    } catch (error) {
      logger.error(`[webhook] 处理错误: ${error.message}`);
      res.statusCode = 500;
      res.end(JSON.stringify({ error: "Internal server error" }));
      return true;
    }
  };
}

// ========================================
// 插件定义
// ========================================

const webhookPlugin = {
  id: "webhook-receiver",
  name: "Webhook Receiver",
  description: "接收外部 Webhook 事件",

  configSchema: {
    parse: (v) => v || {},
    jsonSchema: {
      type: "object",
      additionalProperties: false,
      properties: {
        secret: {
          type: "string",
          description: "Webhook 签名密钥",
        },
      },
    },
    uiHints: {
      secret: {
        label: "签名密钥",
        sensitive: true,
      },
    },
  },

  register(api: OpenClawPluginApi) {
    // 注册 HTTP 路由
    api.registerHttpRoute({
      // 路由路径
      path: "/webhooks/receiver",

      // 认证类型
      // "gateway" - 使用 Gateway 认证
      // "plugin" - 使用插件自己的认证
      auth: "plugin",

      // 匹配模式
      // "exact" - 精确匹配
      // "prefix" - 前缀匹配
      match: "exact",

      // 处理器
      handler: createWebhookHandler(api.logger),
    });

    api.logger.info("[webhook-receiver] 插件已加载");
    api.logger.info("[webhook-receiver] Webhook URL: /plugins/webhook-receiver");
  },
};

export default webhookPlugin;
```

## 5.5 生命周期管理

### 5.5.1 使用生命周期钩子

```typescript
// ========================================
// 使用新的 on() API 注册钩子
// ========================================

register(api: OpenClawPluginApi) {
  // Gateway 启动钩子
  api.on("gateway_start", async (event, ctx) => {
    api.logger.info(`[my-plugin] Gateway 已启动，端口: ${event.port}`);

    // 执行初始化逻辑
    await initializePlugin();
  });

  // Gateway 停止钩子
  api.on("gateway_stop", async (event, ctx) => {
    api.logger.info(`[my-plugin] Gateway 正在停止，原因: ${event.reason ?? "未知"}`);

    // 清理资源
    await cleanupPlugin();
  });

  // Agent 启动前钩子
  api.on("before_agent_start", async (event, ctx) => {
    api.logger.info(`[my-plugin] Agent 即将启动: ${ctx.agentId}`);

    // 可以修改提示词或覆盖模型
    return {
      systemPrompt: "额外系统提示...",
      modelOverride: "special-model",
    };
  });

  // 消息接收钩子
  api.on("message_received", async (event, ctx) => {
    api.logger.info(`[my-plugin] 收到消息: ${event.content}`);

    // 可以记录日志或触发其他操作
  });

  // 消息发送钩子
  api.on("message_sending", async (event, ctx) => {
    api.logger.info(`[my-plugin] 即将发送消息: ${event.content}`);

    // 可以修改或取消消息
    if (shouldBlock(event.content)) {
      return { cancel: true };
    }

    return { content: modifyContent(event.content) };
  });

  // 工具调用前钩子
  api.on("before_tool_call", async (event, ctx) => {
    api.logger.info(`[my-plugin] 工具即将调用: ${event.toolName}`);

    // 可以修改参数或阻止调用
    if (event.toolName === "dangerous_tool") {
      return {
        block: true,
        blockReason: "此工具已被安全策略禁用",
      };
    }

    return {
      params: sanitizeParams(event.params),
    };
  });

  // 工具调用后钩子
  api.on("after_tool_call", async (event, ctx) => {
    api.logger.info(`[my-plugin] 工具调用完成: ${event.toolName}`);

    if (event.error) {
      api.logger.error(`[my-plugin] 工具调用失败: ${event.error}`);
    }
  });
}
```

## 5.7 错误处理最佳实践

### 5.7.1 工具错误处理

```typescript
// ========================================
// 工具错误处理模式
// ========================================

api.registerTool({
  name: "safe_tool",
  description: "带有完善错误处理的工具",
  parameters: Type.Object({
    input: Type.String(),
  }),

  async execute(_toolCallId, params) {
    try {
      // 验证参数
      if (!params?.input) {
        throw new Error("缺少必需参数: input");
      }

      // 执行操作
      const result = await performOperation(params.input);

      // 返回成功结果
      return {
        content: [
          {
            type: "text",
            text: `操作成功: ${result}`,
          },
        ],
      };
    } catch (error) {
      // 区分错误类型
      if (error instanceof ValidationError) {
        // 验证错误 - 用户输入问题
        return {
          content: [
            {
              type: "text",
              text: `输入验证失败: ${error.message}。请检查您的输入。`,
            },
          ],
        };
      }

      if (error instanceof NetworkError) {
        // 网络错误 - 可重试
        return {
          content: [
            {
              type: "text",
              text: `网络错误: ${error.message}。请稍后重试。`,
            },
          ],
        };
      }

      if (error instanceof PermissionError) {
        // 权限错误 - 需要配置
        return {
          content: [
            {
              type: "text",
              text: `权限不足: ${error.message}。请检查插件配置。`,
            },
          ],
        };
      }

      // 未知错误
      api.logger.error(`[safe_tool] 未知错误: ${error}`);
      return {
        content: [
          {
            type: "text",
            text: `操作失败: ${error instanceof Error ? error.message : String(error)}`,
          },
        ],
      };
    }
  },
});
```
