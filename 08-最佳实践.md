### 8.1.3 类型安全

```typescript
// ✅ 推荐：使用 TypeScript 类型定义
interface MyPluginConfig {
  apiKey: string;
  timeout: number;
  retries: number;
}

// 使用类型守卫验证配置
function isMyPluginConfig(value: unknown): value is MyPluginConfig {
  if (typeof value !== "object" || value === null) return false;

  const config = value as Record<string, unknown>;
  return (
    typeof config.apiKey === "string" &&
    typeof config.timeout === "number" &&
    typeof config.retries === "number"
  );
}

function parseConfig(value: unknown): MyPluginConfig {
  if (!isMyPluginConfig(value)) {
    // 返回默认值
    return {
      apiKey: "",
      timeout: 30000,
      retries: 3,
    };
  }
  return value;
}
```

### 8.2.1 提供合理的默认值

```typescript

const DEFAULT_CONFIG = {
  enabled: true,
  timeout: 30000,
  retries: 3,
  logLevel: "info" as const,
};

function parseConfig(value: unknown): MyConfig {
  const raw =
    value && typeof value === "object" && !Array.isArray(value)
      ? (value as Record<string, unknown>)
      : {};

  return {
    enabled: raw.enabled ?? DEFAULT_CONFIG.enabled,
    timeout: raw.timeout ?? DEFAULT_CONFIG.timeout,
    retries: raw.retries ?? DEFAULT_CONFIG.retries,
    logLevel: raw.logLevel ?? DEFAULT_CONFIG.logLevel,
  };
}
```

```typescript
// ✅ 推荐：验证配置范围和格式
function parseConfig(value: unknown): MyConfig {
  const raw =
    value && typeof value === "object" && !Array.isArray(value)
      ? (value as Record<string, unknown>)
      : {};

  // 验证 timeout 范围
  let timeout = Number(raw.timeout) || 30000;
  if (timeout < 1000) timeout = 1000;
  if (timeout > 300000) timeout = 300000;

  // 验证 API Key 格式
  const apiKey = String(raw.apiKey || "");
  if (apiKey && !apiKey.startsWith("sk-")) {
    console.warn("[my-plugin] API Key 格式可能不正确");
  }

  return {
    apiKey,
    timeout,
    retries: Math.max(0, Math.min(10, Number(raw.retries) || 3)),
  };
}
```

## 8.3 错误处理

### 8.3.1 工具错误处理

```typescript
// ✅ 推荐：完善的错误处理
api.registerTool({
  name: "my_tool",
  description: "示例工具",
  parameters: Type.Object({
    input: Type.String(),
  }),

  async execute(_toolCallId, params) {
    try {
      // 参数验证
      if (!params?.input) {
        return {
          content: [
            {
              type: "text" as const,
              text: "错误：请提供 input 参数",
            },
          ],
        };
      }

      // 执行操作
      const result = await performOperation(params.input);

      return {
        content: [
          {
            type: "text" as const,
            text: `操作成功: ${result}`,
          },
        ],
      };
    } catch (error) {
      // 记录详细错误
      api.logger.error(
        `[my_tool] 执行失败: ${error instanceof Error ? error.stack : String(error)}`,
      );

      // 返回用户友好的错误信息
      const userMessage =
        error instanceof ValidationError
          ? `输入验证失败: ${error.message}`
          : error instanceof NetworkError
            ? `网络错误，请稍后重试`
            : `操作失败: ${error instanceof Error ? error.message : "未知错误"}`;

      return {
        content: [
          {
            type: "text" as const,
            text: userMessage,
          },
        ],
      };
    }
  },
});
```

### 8.3.2 自定义错误类型

```typescript
// 定义自定义错误类型
class PluginError extends Error {
  constructor(
    message: string,
    public code: string,
    public recoverable: boolean = true,
  ) {
    super(message);
    this.name = "PluginError";
  }
}

class ValidationError extends PluginError {
  constructor(message: string) {
    super(message, "VALIDATION_ERROR", false);
    this.name = "ValidationError";
  }
}

class NetworkError extends PluginError {
  constructor(message: string, public statusCode?: number) {
    super(message, "NETWORK_ERROR", true);
    this.name = "NetworkError";
  }
}

class RateLimitError extends PluginError {
  constructor(
    message: string,
    public retryAfter: number,
  ) {
    super(message, "RATE_LIMIT_ERROR", true);
    this.name = "RateLimitError";
  }
}

// 使用自定义错误
async function callApi(endpoint: string): Promise<unknown> {
  const response = await fetch(endpoint);

  if (response.status === 429) {
    const retryAfter = Number(response.headers.get("Retry-After")) || 60;
    throw new RateLimitError("API 请求频率超限", retryAfter);
  }

  if (!response.ok) {
    throw new NetworkError(`API 请求失败: ${response.statusText}`, response.status);
  }

  return response.json();
}
```

### 8.4.2 缓存策略

```typescript
// ✅ 推荐：实现缓存
class DataCache {
  private cache = new Map<string, { data: unknown; expires: number }>();
  private ttl: number;

  constructor(ttlMs: number = 60000) {
    this.ttl = ttlMs;
  }

  get(key: string): unknown | undefined {
    const entry = this.cache.get(key);
    if (!entry) return undefined;

    if (Date.now() > entry.expires) {
      this.cache.delete(key);
      return undefined;
    }

    return entry.data;
  }

  set(key: string, data: unknown): void {
    this.cache.set(key, {
      data,
      expires: Date.now() + this.ttl,
    });
  }

  clear(): void {
    this.cache.clear();
  }
}

// 使用缓存
const cache = new DataCache(5 * 60 * 1000); // 5 分钟 TTL

async function getDataWithCache(key: string): Promise<unknown> {
  // 检查缓存
  const cached = cache.get(key);
  if (cached !== undefined) {
    return cached;
  }

  // 获取数据
  const data = await fetchData(key);

  // 存入缓存
  cache.set(key, data);

  return data;
}
```

### 8.5.4 速率限制

```typescript
// ✅ 推荐：实现速率限制
class RateLimiter {
  private requests = new Map<string, number[]>();
  private limit: number;
  private windowMs: number;

  constructor(limit: number = 10, windowMs: number = 60000) {
    this.limit = limit;
    this.windowMs = windowMs;
  }

  isAllowed(key: string): boolean {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // 获取或初始化请求列表
    let requests = this.requests.get(key) || [];

    // 清理过期请求
    requests = requests.filter((t) => t > windowStart);

    // 检查是否超限
    if (requests.length >= this.limit) {
      return false;
    }

    // 记录新请求
    requests.push(now);
    this.requests.set(key, requests);

    return true;
  }
}

// 使用速率限制
const limiter = new RateLimiter(10, 60000); // 每分钟 10 次

api.registerTool({
  name: "api_call",
  // ...
  async execute(_toolCallId, params) {
    const userId = params?.userId || "anonymous";

    if (!limiter.isAllowed(userId)) {
      return {
        content: [
          {
            type: "text" as const,
            text: "请求过于频繁，请稍后再试",
          },
        ],
      };
    }

    // 执行操作...
  },
});
```
