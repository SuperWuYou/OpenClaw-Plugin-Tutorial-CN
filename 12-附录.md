## 12.6 工具定义模板

```typescript
import { Type } from "@sinclair/typebox";

const myTool = {
  name: "my_tool",
  label: "My Tool",
  description: "工具描述，Agent 根据此描述决定是否调用",

  parameters: Type.Object({
    // 必需参数
    required_param: Type.String({
      description: "参数描述",
    }),

    // 可选参数
    optional_param: Type.Optional(
      Type.String({
        description: "可选参数描述",
      }),
    ),

    // 带默认值的参数
    param_with_default: Type.Optional(
      Type.Number({
        description: "带默认值的参数",
        default: 10,
      }),
    ),

    // 枚举参数
    enum_param: Type.Optional(
      Type.String({
        description: "枚举参数",
        enum: ["option1", "option2", "option3"],
      }),
    ),

    // 数组参数
    array_param: Type.Optional(
      Type.Array(Type.String(), {
        description: "数组参数",
      }),
    ),
  }),

  async execute(_toolCallId: string, params: Record<string, unknown>) {
    try {
      // 获取参数
      const requiredParam = params?.required_param as string;
      const optionalParam = params?.optional_param as string | undefined;
      const paramWithDefault = (params?.param_with_default as number) ?? 10;

      // 验证参数
      if (!requiredParam) {
        return {
          content: [{ type: "text" as const, text: "错误：缺少必需参数" }],
        };
      }

      // 执行操作
      const result = await performOperation(requiredParam, optionalParam);

      // 返回结果
      return {
        content: [
          {
            type: "text" as const,
            text: `操作成功: ${result}`,
          },
        ],
        details: { result },
      };
    } catch (error) {
      return {
        content: [
          {
            type: "text" as const,
            text: `操作失败: ${error instanceof Error ? error.message : String(error)}`,
          },
        ],
      };
    }
  },
};
```

### 12.8.2 带超时的请求

```typescript
async function fetchWithTimeout(
  url: string,
  options: RequestInit,
  timeout: number,
): Promise<Response> {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeout);

  try {
    const response = await fetch(url, {
      ...options,
      signal: controller.signal,
    });
    return response;
  } finally {
    clearTimeout(timeoutId);
  }
}
```

### 12.8.3 重试逻辑

```typescript
async function withRetry<T>(
  fn: () => Promise<T>,
  retries: number = 3,
  delay: number = 1000,
): Promise<T> {
  let lastError: Error | undefined;

  for (let i = 0; i < retries; i++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error instanceof Error ? error : new Error(String(error));
      if (i < retries - 1) {
        await new Promise((resolve) => setTimeout(resolve, delay));
      }
    }
  }

  throw lastError;
}
```

### 12.8.4 速率限制

```typescript
class RateLimiter {
  private requests = new Map<string, number[]>();

  check(key: string, limit: number, windowMs: number): boolean {
    const now = Date.now();
    const requests = this.requests.get(key) || [];

    // 清理过期请求
    const validRequests = requests.filter((t) => now - t < windowMs);

    if (validRequests.length >= limit) {
      return false;
    }

    validRequests.push(now);
    this.requests.set(key, validRequests);
    return true;
  }
}
```
